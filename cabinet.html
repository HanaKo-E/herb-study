// 修改後的生成函式 function createCabinet(containerId, start, end) { const container = document.getElementById(containerId); container.innerHTML = ""; // 清空容器 for (let i = start; i <= end; i++) { const id = i.toString(); const name = medicineMap[id]; const div = document.createElement('div'); // 如果是空號，套用 gray-50 背景；若有藥材，套用白色背景 div.className = `drawer ${name ? 'bg-white' : 'bg-gray-50 border-dashed text-gray-400'}`; // 核心邏輯：若 name 為空、undefined 或為空字串，則顯示 " - " const displayName = (name && name.trim() !== "") ? name : " - "; div.innerHTML = ` ${id} ${displayName} `; // 僅在有藥材時才綁定點擊跳轉事件 if (name && name !== " - ") { div.onclick = () => { location.href = `index.html?search=${encodeURIComponent(name)}`; }; div.classList.add('cursor-pointer', 'hover:bg-green-50'); } container.appendChild(div); } }
